# Cài đặt Git LFS
git lfs install

# Theo dõi các tệp .pkl bằng Git LFS
git lfs track "*.pkl"

# Thêm và commit lại các tệp lớn
git add .gitattributes

git add .

git commit -m "Using Git LFS"

# Đẩy các thay đổi lên remote repository:
git branch -M main

git push -u origin main
